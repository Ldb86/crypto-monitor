//@version=5
strategy("📊 MARADONA Brain + PELÉ Execution V18.3.8 CLEAN ALERTS", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=10, max_boxes_count=150, max_lines_count=500, max_labels_count=500)


// ================= VERSIONE ================= //
// V17 = operativa: MACD scout + Breakout Box 20 + Micro Golden Zone/FVG.
// Focus: entry operative, test micro GZ, senza appesantire Pine.

// ================= INPUT STRATEGY ================= //
capitalInput = input.float(1000, "💰 Capitale per trade")
riskUSD = input.float(10, "🛑 SL massimo $ per trade")
adxLen = input.int(14, "Periodo ADX")
adxThresh = input.float(20.0, "Soglia ADX")
volLen = input.int(20, "Media Volume")
lookback = input.int(20, "Breakout Box")
targetMult = input.float(1.0, "TP Mult")
stopMult = input.float(0.5, "SL Mult")
entryMode = input.string(defval="Pullback Retail + Smart", title="Modalita Entry", options=["Pullback Retail + Smart", "Solo Retail", "Solo Istituzionale", "MACD Classico", "Solo Visual"])
maxRetestBars = input.int(20, "⏳ Barre max attesa retest", minval=1)
gzMidFib = input.float(0.618, "🎯 Golden Zone media", minval=0.55, maxval=0.705, step=0.001)
gzDeepFib = input.float(0.705, "🎯 Golden Zone profonda", minval=0.618, maxval=0.886, step=0.001)
entryAfterSweepReclaim = input.bool(true, "🎯 Entry dopo sweep + reclaim + pullback GZ")

// ================= V18 TIMEFRAME ROUTER ================= //
useV18UnifiedMode = input.bool(true, "🧠 V18 Unificato: Pelé sotto 4H / Maradona da 4H+")
v18HTFMinutes = input.int(240, "🧠 Soglia HTF minuti", minval=60)
v18IsHTF = useV18UnifiedMode and (not timeframe.isintraday or timeframe.multiplier >= v18HTFMinutes)
v18IsLTF = not v18IsHTF
v18ModeTxt = v18IsHTF ? "MARADONA HTF" : "PELÉ LTF"

// ================= V18.1 ENTRY AUTH ENGINE ================= //
useV181EntryAuth = input.bool(true, "⚔️ V18.1 Entry: MACD + TL Break")
useV182DualPrealert = input.bool(true, "⚔️ V18.3.6 MACD/TL prealert + seconda condizione = entry")
showMicroZonePrices = input.bool(true, "🏷️ Mostra prezzi Micro GZ")
showMicroTPLevels = input.bool(true, "🎯 Mostra TP/SL Micro GZ")
microTP1Mult = input.float(0.5, "🎯 Micro TP1 range mult", minval=0.1, step=0.1)
microTP2Mult = input.float(1.0, "🎯 Micro TP2 range mult", minval=0.1, step=0.1)
microTP3Mult = input.float(1.5, "🎯 Micro TP3 range mult", minval=0.1, step=0.1)



useOperationalClassicTrigger = input.bool(true, "⚔️ Entry operative MACD + Breakout Box 20")
useMicroGZAfterTrigger = input.bool(true, "🎯 Micro Golden Zone dopo trigger MACD/Box")
microGZBars = input.int(60, "🎯 Durata Micro Golden Zone", minval=5, maxval=300)
microGZUseFVG = input.bool(true, "🎯 Micro GZ usa FVG se presente")

// ================= TOGGLE VISUAL ================= //
showTradeLabels = input.bool(false, "Mostra etichette entry")
showQtyLabels = input.bool(false, "Mostra quantità a video")
showPivots = input.bool(true, "📍 Mostra Pivot Levels")
showZones = input.bool(true, "🟦 Mostra Zone Pivot")
showColor = input.bool(true, "🎨 Colori dinamici")
showIchimoku = input.bool(true, "☁️ Mostra Ichimoku")
showTrendline = input.bool(true, "📐 Mostra Trendline")
showRetailZone = input.bool(true, "🔵🟠 Mostra Golden Zone Retail")
showTradeOriginZone = input.bool(true, "📦 Mostra Zona Origine Trade")

// ================= MACD ================= //
[macdLine, signalLine, _] = ta.macd(close, 26, 50, 9)
macdCrossUp = ta.crossover(macdLine, signalLine)
macdCrossDown = ta.crossunder(macdLine, signalLine)

// ================= PRE-ALERT INPUT ================= //
showPreAlerts = input.bool(true, "👁 Mostra PRE/RADAR su grafico")
enablePreAlertAlerts = input.bool(false, "🔕 Invia notifiche PRE legacy (OFF consigliato)")


// ================= MACD PRE-ALERT STATE ENGINE V13 — ANTI CHOP / SEARCH MODE ================= //
// MACD non è entry: accende il radar.
// Affinato per non impazzire in lateralizzazione: flip frequenti, range stretto, ADX scarico.
useMacdStateEngine = input.bool(true, "🧠 MACD State Engine anti-lateralizzazione")
macdResetVotesNeeded = input.int(1, "🧠 MACD reset: voti minimi cambio trend", minval=1, maxval=5)
macdMinDistance = input.float(0.0, "🧠 MACD distanza minima", step=0.0001)
macdSearchBars = input.int(80, "🧠 Barre ricerca trade dopo preallarme", minval=5, maxval=500)

macdChopLookback = input.int(18, "🧠 MACD anti-chop lookback", minval=5, maxval=100)
macdMaxCrossInChop = input.int(5, "🧠 Max incroci MACD in chop", minval=1, maxval=10)
macdUseADXChop = input.bool(true, "🧠 Blocca MACD se ADX basso/range")
macdUseBoxBreak = input.bool(false, "🧠 MACD valido con breakout box 20")
macdAllowEarlyHTF = input.bool(true, "🧠 MACD early se HTF forte")

macdDistanceOK = math.abs(macdLine - signalLine) > macdMinDistance

miniTrendBull = close > ta.ema(close, 21) and ta.ema(close, 21) > ta.ema(close, 50)
miniTrendBear = close < ta.ema(close, 21) and ta.ema(close, 21) < ta.ema(close, 50)

rangeBreakLong20 = close > ta.highest(high, 20)[1]
rangeBreakShort20 = close < ta.lowest(low, 20)[1]

macdCrossAny = macdCrossUp or macdCrossDown
macdCrossCount = math.sum(macdCrossAny ? 1 : 0, macdChopLookback)
macdChoppy = macdCrossCount > macdMaxCrossInChop

macdBoxRange = ta.highest(high, 20) - ta.lowest(low, 20)
macdBoxRangePrev = ta.highest(high[20], 20) - ta.lowest(low[20], 20)
macdBoxCompressed = macdBoxRangePrev > 0 and macdBoxRange < macdBoxRangePrev * 0.70

macdADXChop = macdUseADXChop and macdBoxCompressed
macdNoChop = not macdChoppy and not macdADXChop

macdHTFLongPermission = macdAllowEarlyHTF and close > ta.ema(close, 50)
macdHTFShortPermission = macdAllowEarlyHTF and close < ta.ema(close, 50)

macdBreakLongOK = not macdUseBoxBreak or rangeBreakLong20 or macdHTFLongPermission
macdBreakShortOK = not macdUseBoxBreak or rangeBreakShort20 or macdHTFShortPermission

macdResetLongVotes =
     (miniTrendBull ? 1 : 0) +
     (rangeBreakLong20 ? 1 : 0) +
     (macdLine > 0 ? 1 : 0) +
     (macdHTFLongPermission ? 1 : 0)

macdResetShortVotes =
     (miniTrendBear ? 1 : 0) +
     (rangeBreakShort20 ? 1 : 0) +
     (macdLine < 0 ? 1 : 0) +
     (macdHTFShortPermission ? 1 : 0)

macdRealLongShift = macdResetLongVotes >= macdResetVotesNeeded
macdRealShortShift = macdResetShortVotes >= macdResetVotesNeeded

var int macdState = 0
var int macdPreAlertBar = na
var bool macdSearchLong = false
var bool macdSearchShort = false
var int macdLastValidCrossBar = na

macdLongPreAlert = useMacdStateEngine ? (macdCrossUp and macdState != 1 and macdDistanceOK and macdNoChop and macdBreakLongOK and macdRealLongShift) : macdCrossUp
macdShortPreAlert = useMacdStateEngine ? (macdCrossDown and macdState != -1 and macdDistanceOK and macdNoChop and macdBreakShortOK and macdRealShortShift) : macdCrossDown

if macdLongPreAlert
    macdState := 1
    macdSearchLong := true
    macdSearchShort := false
    macdPreAlertBar := bar_index
    macdLastValidCrossBar := bar_index

if macdShortPreAlert
    macdState := -1
    macdSearchShort := true
    macdSearchLong := false
    macdPreAlertBar := bar_index
    macdLastValidCrossBar := bar_index

macdSearchExpired = not na(macdPreAlertBar) and bar_index - macdPreAlertBar > macdSearchBars
if macdSearchExpired
    macdSearchLong := false
    macdSearchShort := false

// Cambio stato solo con shift reale, non con ogni flip laterale.
if macdState == 1 and macdCrossDown and macdNoChop and macdRealShortShift and macdBreakShortOK
    macdState := -1
    macdSearchShort := true
    macdSearchLong := false
    macdPreAlertBar := bar_index
    macdLastValidCrossBar := bar_index

if macdState == -1 and macdCrossUp and macdNoChop and macdRealLongShift and macdBreakLongOK
    macdState := 1
    macdSearchLong := true
    macdSearchShort := false
    macdPreAlertBar := bar_index
    macdLastValidCrossBar := bar_index

macdStateTxt =
     macdState == 1 ? (macdChoppy ? "MACD L CHOP" : "MACD LONG") :
     macdState == -1 ? (macdChoppy ? "MACD S CHOP" : "MACD SHORT") :
     macdChoppy ? "MACD CHOP" :
     "MACD NEUTRO"



// ================= ADX ================= //
[_, _, adx] = ta.dmi(adxLen, adxLen)
adxOK = adx >= adxThresh

// ================= VOLUME ================= //
vol = volume
volMA = ta.sma(vol, volLen)
volSTD = ta.stdev(vol, volLen)
volSignificativo = vol > volMA + volSTD * 0.5

// ================= BOX ================= //
boxHigh = ta.highest(high[1], lookback)
boxLow = ta.lowest(low[1], lookback)
boxRng = boxHigh - boxLow

// ================= V18.2.5 TL BREAK EARLY DEFINITIONS ================= //
// Definite subito dopo il BOX 20, prima di monitor/prealert/entry.
v181TlBreakLong = close > ta.highest(high, 10)[1] or close > boxHigh
v181TlBreakShort = close < ta.lowest(low, 10)[1] or close < boxLow



// ================= MODALITÀ ================= //
useClassic = entryMode == "MACD Classico"
useRetail = entryMode == "Pullback Retail + Smart" or entryMode == "Solo Retail"
useSmart = entryMode == "Pullback Retail + Smart" or entryMode == "Solo Istituzionale"
onlyVisual = entryMode == "Solo Visual"

// ================= FRACTAL BOS / CHOCH STRUCTURE ENGINE ================= //
showStructure = input.bool(true, "🏛️ Mostra BOS / CHoCH Fractal")
useStructureFilter = input.bool(false, "🎯 Usa BOS/CHoCH come filtro entry")
structureLen = input.int(5, "Swing Fractal BOS/CHoCH", minval=3)
showStructureLines = input.bool(true, "📏 Mostra linee BOS/CHoCH")
showBrokenSR = input.bool(false, "📍 Mostra Support/Resistance rotti")

bullCss = color.new(color.green, 0)
bearCss = color.new(color.red, 0)
p = int(structureLen / 2)
nStruct = bar_index

dh = math.sum(math.sign(high - high[1]), p)
dl = math.sum(math.sign(low - low[1]), p)
bullFractal = dh == -p and dh[p] == p and high[p] == ta.highest(high, structureLen)
bearFractal = dl == p and dl[p] == -p and low[p] == ta.lowest(low, structureLen)

var float upperValue = na
var int upperLoc = na
var bool upperCrossed = false
var float lowerValue = na
var int lowerLoc = na
var bool lowerCrossed = false
var int structureTrend = 0

if bullFractal
    upperValue := high[p]
    upperLoc := nStruct - p
    upperCrossed := false

if bearFractal
    lowerValue := low[p]
    lowerLoc := nStruct - p
    lowerCrossed := false

bosLong = false
bosShort = false
chochLong = false
chochShort = false

if not na(upperValue) and ta.crossover(close, upperValue) and not upperCrossed
    chochLong := structureTrend == -1
    bosLong := structureTrend != -1
    if showStructureLines
        line.new(upperLoc, upperValue, nStruct, upperValue, color=showStructure ? bullCss : na, width=1)
    if showStructure
        label.new(int(math.avg(nStruct, upperLoc)), upperValue, chochLong ? "CHoCH" : "BOS", style=label.style_label_down, color=color.new(color.white, 100), textcolor=bullCss, size=size.tiny)
    upperCrossed := true
    structureTrend := 1

if not na(lowerValue) and ta.crossunder(close, lowerValue) and not lowerCrossed
    chochShort := structureTrend == 1
    bosShort := structureTrend != 1
    if showStructureLines
        line.new(lowerLoc, lowerValue, nStruct, lowerValue, color=showStructure ? bearCss : na, width=1)
    if showStructure
        label.new(int(math.avg(nStruct, lowerLoc)), lowerValue, chochShort ? "CHoCH" : "BOS", style=label.style_label_up, color=color.new(color.white, 100), textcolor=bearCss, size=size.tiny)
    lowerCrossed := true
    structureTrend := -1

var line lastSupportLine = na
var line lastResistanceLine = na

if showBrokenSR and bearFractal
    line.delete(lastSupportLine)
    lastSupportLine := line.new(lowerLoc, lowerValue, nStruct, lowerValue, color=color.new(color.green, 40), style=line.style_dashed)

if showBrokenSR and bullFractal
    line.delete(lastResistanceLine)
    lastResistanceLine := line.new(upperLoc, upperValue, nStruct, upperValue, color=color.new(color.red, 40), style=line.style_dashed)

if showBrokenSR and not na(lastSupportLine)
    line.set_x2(lastSupportLine, nStruct)

if showBrokenSR and not na(lastResistanceLine)
    line.set_x2(lastResistanceLine, nStruct)

structureLongOK = not useStructureFilter or bosLong or chochLong
structureShortOK = not useStructureFilter or bosShort or chochShort

// ================= FASE 2 — FVG + HTF BIAS ================= //
useFVGFilter = input.bool(false, "🧩 Usa FVG come filtro entry")
showFVG = input.bool(false, "🟩🟥 Mostra Fair Value Gap")
maxFVGBoxes = input.int(5, "Numero max FVG visibili", minval=1, maxval=20)
useHTFBias = input.bool(false, "🧭 Usa HTF Bias come filtro entry")
htfTf = input.timeframe("60", "Timeframe HTF Bias")

htfClose = request.security(syminfo.tickerid, htfTf, close)
htfEmaFast = request.security(syminfo.tickerid, htfTf, ta.ema(close, 21))
htfEmaSlow = request.security(syminfo.tickerid, htfTf, ta.ema(close, 50))
htfBull = htfClose > htfEmaFast and htfEmaFast > htfEmaSlow
htfBear = htfClose < htfEmaFast and htfEmaFast < htfEmaSlow
htfBiasTxt = htfBull ? "HTF BULL" : htfBear ? "HTF BEAR" : "HTF NEUTRO"

bullFVG = low > high[2]
bearFVG = high < low[2]
bullFVGTop = low
bullFVGBottom = high[2]
bearFVGTop = low[2]
bearFVGBottom = high

var box[] bullFVGBoxes = array.new_box()
var box[] bearFVGBoxes = array.new_box()

if showFVG and bullFVG
    newBullBox = box.new(bar_index - 2, bullFVGTop, bar_index + 30, bullFVGBottom, bgcolor=color.new(color.green, 85), border_color=color.new(color.green, 20), text="BULL FVG", text_color=color.green)
    array.push(bullFVGBoxes, newBullBox)
    if array.size(bullFVGBoxes) > maxFVGBoxes
        oldBullBox = array.shift(bullFVGBoxes)
        box.delete(oldBullBox)

if showFVG and bearFVG
    newBearBox = box.new(bar_index - 2, bearFVGTop, bar_index + 30, bearFVGBottom, bgcolor=color.new(color.red, 85), border_color=color.new(color.red, 20), text="BEAR FVG", text_color=color.red)
    array.push(bearFVGBoxes, newBearBox)
    if array.size(bearFVGBoxes) > maxFVGBoxes
        oldBearBox = array.shift(bearFVGBoxes)
        box.delete(oldBearBox)

var float lastBullFVGTop = na
var float lastBullFVGBottom = na
var float lastBearFVGTop = na
var float lastBearFVGBottom = na
if bullFVG
    lastBullFVGTop := bullFVGTop
    lastBullFVGBottom := bullFVGBottom
if bearFVG
    lastBearFVGTop := bearFVGTop
    lastBearFVGBottom := bearFVGBottom

priceInBullFVG = not na(lastBullFVGTop) and close <= lastBullFVGTop and close >= lastBullFVGBottom
priceInBearFVG = not na(lastBearFVGTop) and close <= lastBearFVGTop and close >= lastBearFVGBottom
fvgLongOK = not useFVGFilter or priceInBullFVG
fvgShortOK = not useFVGFilter or priceInBearFVG
htfLongOK = not useHTFBias or htfBull
htfShortOK = not useHTFBias or htfBear

// ================= FASE 3 — SESSION ENGINE / KILLZONE ================= //
useSessionFilter = input.bool(false, "🕒 Usa filtro sessioni Killzone")
showSessionBg = input.bool(false, "🎨 Mostra sfondo sessioni")
sessionTimezone = input.string("Europe/Rome", "Timezone Sessioni")
londonSession = input.session("0800-1100", "🇬🇧 London Killzone")
nySession = input.session("1430-1700", "🇺🇸 New York Killzone")
overlapSession = input.session("1430-1600", "⚡ London/NY Overlap")
inLondon = not na(time(timeframe.period, londonSession, sessionTimezone))
inNY = not na(time(timeframe.period, nySession, sessionTimezone))
inOverlap = not na(time(timeframe.period, overlapSession, sessionTimezone))
sessionOK = not useSessionFilter or inLondon or inNY or inOverlap
sessionTxt = inOverlap ? "OVERLAP" : inLondon ? "LONDON" : inNY ? "NY" : "OFF"

// ================= VAR TRADE ================= //
var float tp = na
var float sl = na
var float entryPrice = na
var float qtyCalc = na
inTrade = strategy.position_size != 0

// ================= MACD CLASSICO SOLO SE SELEZIONATO ================= //
if useClassic and not onlyVisual and macdCrossUp and strategy.position_size == 0
    sl := close - boxRng * stopMult
    tp := close + boxRng * targetMult
    entryPrice := close
    qtyCalc := riskUSD / math.max(math.abs(close - sl), syminfo.mintick)
    strategy.entry("LONG CLASSIC", strategy.long, qty=qtyCalc)
    strategy.exit("TP SL LONG CLASSIC", from_entry="LONG CLASSIC", limit=tp, stop=sl)

if useClassic and not onlyVisual and macdCrossDown and strategy.position_size == 0
    sl := close + boxRng * stopMult
    tp := close - boxRng * targetMult
    entryPrice := close
    qtyCalc := riskUSD / math.max(math.abs(close - sl), syminfo.mintick)
    strategy.entry("SHORT CLASSIC", strategy.short, qty=qtyCalc)
    strategy.exit("TP SL SHORT CLASSIC", from_entry="SHORT CLASSIC", limit=tp, stop=sl)

// ================= RETAIL PULLBACK GOLDEN ZONE ================= //
var bool retailLongActive = false
var bool retailShortActive = false
var int retailSignalBar = na
var float retailHigh = na
var float retailLow = na

if useRetail and macdLongPreAlert
    retailHigh := close
    retailLow := boxLow
    retailLongActive := true
    retailShortActive := false
    retailSignalBar := bar_index

if useRetail and macdShortPreAlert
    retailLow := close
    retailHigh := boxHigh
    retailShortActive := true
    retailLongActive := false
    retailSignalBar := bar_index

retailExpired = not na(retailSignalBar) and bar_index - retailSignalBar > maxRetestBars
if retailExpired
    retailLongActive := false
    retailShortActive := false

retailValid = not na(retailHigh) and not na(retailLow) and retailHigh > retailLow
retailFib50Long = retailValid ? retailHigh - (retailHigh - retailLow) * 0.5 : na
retailFib618MidLong = retailValid ? retailHigh - (retailHigh - retailLow) * gzMidFib : na
retailFib618Long = retailValid ? retailHigh - (retailHigh - retailLow) * gzDeepFib : na
retailFib50Short = retailValid ? retailLow + (retailHigh - retailLow) * 0.5 : na
retailFib618MidShort = retailValid ? retailLow + (retailHigh - retailLow) * gzMidFib : na
retailFib618Short = retailValid ? retailLow + (retailHigh - retailLow) * gzDeepFib : na
retailInZoneLong = retailLongActive and close <= retailFib50Long and close >= retailFib618Long
retailInZoneShort = retailShortActive and close >= retailFib50Short and close <= retailFib618Short
retailConfirmLong = close > open and close > close[1]
retailConfirmShort = close < open and close < close[1]
retailEntryLong = retailValid ? math.avg(retailFib50Long, retailFib618Long) : na
retailEntryShort = retailValid ? math.avg(retailFib50Short, retailFib618Short) : na


// ================= V18 LTF MACD PREALERT MONITOR ================= //
v18UseLTFMacdPrealerts = input.bool(true, "🔔 V18 LTF: alert MACD preallarme")
v18MonitorBars = input.int(80, "🔔 Barre monitoraggio dopo MACD", minval=10, maxval=300)

v18MacdPreLong = v18IsLTF and v18UseLTFMacdPrealerts and (macdCrossUp or (useV182DualPrealert and v181TlBreakLong))
v18MacdPreShort = v18IsLTF and v18UseLTFMacdPrealerts and (macdCrossDown or (useV182DualPrealert and v181TlBreakShort))

var int v18MonitorDir = 0
var int v18MonitorBar = na

if v18MacdPreLong
    v18MonitorDir := 1
    v18MonitorBar := bar_index

if v18MacdPreShort
    v18MonitorDir := -1
    v18MonitorBar := bar_index

if not na(v18MonitorBar) and bar_index - v18MonitorBar > v18MonitorBars
    v18MonitorDir := 0

v18MonitorLong = v18IsLTF and v18MonitorDir == 1
v18MonitorShort = v18IsLTF and v18MonitorDir == -1
v18MonitorTxt = v18MonitorLong ? "MONITOR LONG" : v18MonitorShort ? "MONITOR SHORT" : v18IsHTF ? "HTF BRAIN" : "WAIT"

// ================= V18.2.5 DUAL PREALERT + AUTH CORE CLEAN ================= //
// MACD o TL = preallarme e crea Micro GZ.
// Entry autorizzata solo quando entrambe le condizioni sono viste dentro la finestra monitor.
v182MacdLongSignal = v18IsLTF and macdCrossUp
v182MacdShortSignal = v18IsLTF and macdCrossDown
v182TlLongSignal = v18IsLTF and v181TlBreakLong
v182TlShortSignal = v18IsLTF and v181TlBreakShort

v182PreLong = useV182DualPrealert and (v182MacdLongSignal or v182TlLongSignal)
v182PreShort = useV182DualPrealert and (v182MacdShortSignal or v182TlShortSignal)

var bool v182MacdLongSeen = false
var bool v182TlLongSeen = false
var bool v182MacdShortSeen = false
var bool v182TlShortSeen = false
var int v182LongPreBar = na
var int v182ShortPreBar = na

if v182PreLong
    v182MacdLongSeen := v182MacdLongSeen or v182MacdLongSignal
    v182TlLongSeen := v182TlLongSeen or v182TlLongSignal
    v182MacdShortSeen := false
    v182TlShortSeen := false
    v182LongPreBar := bar_index

if v182PreShort
    v182MacdShortSeen := v182MacdShortSeen or v182MacdShortSignal
    v182TlShortSeen := v182TlShortSeen or v182TlShortSignal
    v182MacdLongSeen := false
    v182TlLongSeen := false
    v182ShortPreBar := bar_index

if not na(v182LongPreBar) and bar_index - v182LongPreBar > v18MonitorBars
    v182MacdLongSeen := false
    v182TlLongSeen := false

if not na(v182ShortPreBar) and bar_index - v182ShortPreBar > v18MonitorBars
    v182MacdShortSeen := false
    v182TlShortSeen := false

v182LongSecondCondition = v182MacdLongSeen and v182TlLongSeen
v182ShortSecondCondition = v182MacdShortSeen and v182TlShortSeen

v181LongAuthorized = v18IsLTF and (useV182DualPrealert ? v182LongSecondCondition : (v18MonitorLong and v181TlBreakLong))
v181ShortAuthorized = v18IsLTF and (useV182DualPrealert ? v182ShortSecondCondition : (v18MonitorShort and v181TlBreakShort))

v181AuthTxt =
     v181LongAuthorized ? "AUTH LONG" :
     v181ShortAuthorized ? "AUTH SHORT" :
     v182MacdLongSeen and not v182TlLongSeen ? "PRE L MACD WAIT TL" :
     v182TlLongSeen and not v182MacdLongSeen ? "PRE L TL WAIT MACD" :
     v182MacdShortSeen and not v182TlShortSeen ? "PRE S MACD WAIT TL" :
     v182TlShortSeen and not v182MacdShortSeen ? "PRE S TL WAIT MACD" :
     v18MonitorTxt

v182StateTxt = v181AuthTxt



v18TLBullVote = 0
v18TLBearVote = 0

// ================= OPERATIONAL MICRO GZ V16.2 — MACD + BOX 20 ================= //
// Questa è la parte operativa: MACD accende, breakout box 20 conferma,

mtfCore(_tfWeight) =>
    _c = close
    _o = open
    _h = high
    _l = low
    _v = volume
    _atr = ta.atr(14)
    _ema21 = ta.ema(close, 21)
    _ema50 = ta.ema(close, 50)
    _ema200 = ta.ema(close, 200)
    _volMA = ta.sma(volume, 20)
    _sHigh = ta.highest(high[1], 20)
    _sLow = ta.lowest(low[1], 20)
    _range = math.max(_sHigh - _sLow, syminfo.mintick)
    _mid = math.avg(_sHigh, _sLow)

    _trendBull = _c > _ema21 and _ema21 > _ema50 and _ema50 > _ema200
    _trendBear = _c < _ema21 and _ema21 < _ema50 and _ema50 < _ema200
    _softBull = _c > _ema21 and _ema21 > _ema50
    _softBear = _c < _ema21 and _ema21 < _ema50

    _bosL = _c > _sHigh
    _bosS = _c < _sLow
    _chochL = _c > _mid and _c[1] < _mid
    _chochS = _c < _mid and _c[1] > _mid

    _sweepL = _l < _sLow and _c > _sLow
    _sweepS = _h > _sHigh and _c < _sHigh

    _fib50L = _sHigh - _range * 0.5
    _fib618L = _sHigh - _range * 0.618
    _fib50S = _sLow + _range * 0.5
    _fib618S = _sLow + _range * 0.618
    _gzL = _c <= _fib50L and _c >= _fib618L
    _gzS = _c >= _fib50S and _c <= _fib618S

    _entryL = math.avg(_fib50L, _fib618L)
    _entryS = math.avg(_fib50S, _fib618S)
    _slL = _sLow - _atr * 0.20
    _slS = _sHigh + _atr * 0.20

    _tpL1 = _entryL + math.abs(_entryL - _slL) * 1.0
    _tpL2 = _entryL + math.abs(_entryL - _slL) * 1.6
    _tpL3 = _entryL + math.abs(_entryL - _slL) * 2.2
    _tpS1 = _entryS - math.abs(_slS - _entryS) * 1.0
    _tpS2 = _entryS - math.abs(_slS - _entryS) * 1.6
    _tpS3 = _entryS - math.abs(_slS - _entryS) * 2.2

    _rrL = math.abs(_tpL2 - _entryL) / math.max(math.abs(_entryL - _slL), syminfo.mintick)

         _trendBull ? 1 :
